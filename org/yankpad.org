        let newConfig = this.pageapi.mainWindow.adjustWindowConfig({
            width: Screen.widthPixels,
            height: Screen.heightPixels,
            x: 0,
            y: 0
        });

        return {
            left: newConfig.x,
            top: newConfig.y + STATUSBAR_HEIGHT,
            width: newConfig.width,
            height: newConfig.height - STATUSBAR_HEIGHT,
            type: 5,
            opaque: false
        };
    }
